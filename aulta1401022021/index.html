    • CREATE DATABASE cad_user
      /SQL no Postgres - Criando Tabelas CREATE TABLE cad_user (
      id SERIAL PRIMARY KEY,
      nome VARCHAR (50) NOT NULL,
      telefonefixo VARCHAR (20),
      celular VARCHAR (20),
      email VARCHAR (50) NOT NULL
      );

              1. --INSTRUÇÕES SELECT

              --Comando ORDER BY
